Baixar PDF do Google Drive (Somente Leitura)

Baixe documentos que só podem ser visualizados — o script junta tudo em um único PDF

Use apenas em documentos que você tem permissão legítima para visualizar. O conteúdo já está visível na sua tela — este script apenas salva localmente.
Método 1: Console do Navegador
Método 2: Botão na Barra de Favoritos
1

Abra o PDF no Google Drive

Acesse o link do PDF e espere ele carregar por completo na tela
2

Abra o Console do navegador

Pressione a tecla F12 no teclado. Vai abrir um painel na tela. Clique na aba Console.
3

Copie o script abaixo e cole no Console

Clique no botão "Copiar", depois clique dentro do Console e pressione Ctrl+V 
(async function(){
  /* Carrega a biblioteca jsPDF para gerar o PDF final */
  if(!window.jspdf){
    const s=document.createElement('script');
    s.src='https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js';
    document.head.appendChild(s);
    await new Promise(r=>s.onload=r);
  }
  const{jsPDF}=window.jspdf;

  const urls=new Set();
  const sleep=ms=>new Promise(r=>setTimeout(r,ms));

  function collect(){
    document.querySelectorAll('img').forEach(img=>{
      const s=img.src||'';
      if((s.includes('googleusercontent')||s.includes('lh3')||s.includes('ggpht'))&&img.naturalWidth>200){
        urls.add(s);
      }
    });
    document.querySelectorAll('[style*="background"]').forEach(el=>{
      const bg=getComputedStyle(el).backgroundImage;
      if(bg&&bg.includes('url(')){
        const m=bg.match(/url\(["']?(.*?)["']?\)/);
        if(m&&(m[1].includes('googleusercontent')||m[1].includes('lh3')))urls.add(m[1]);
      }
    });
  }

  const container=document.querySelector('.ndfHFb-c4YZDc-q77wJc')
    ||document.querySelector('[role="document"]')
    ||document.body;
  let lastH=0;

  console.log('Rolando o documento... nao feche esta aba.');
  for(let i=0;i<300;i++){
    collect();
    container.scrollBy(0,window.innerHeight*0.8);
    await sleep(700);
    if(container.scrollHeight===lastH&&i>3)break;
    lastH=container.scrollHeight;
  }

  container.scrollTo(0,0);
  await sleep(500);
  collect();

  const arr=[...urls].sort();
  console.log('Encontradas '+arr.length+' paginas. Baixando e montando o PDF...');

  const pdf=new jsPDF({unit:'px',hotfixes:{px_scaling:1}});
  let first=true;

  for(let i=0;ir.blob());
      const dataUrl=await new Promise((res,rej)=>{
        const fr=new FileReader();
        fr.onload=()=>res(fr.result);
        fr.onerror=rej;
        fr.readAsDataURL(blob);
      });
      const img=new Image();
      img.src=dataUrl;
      await new Promise(r=>img.onload=r);

      const w=img.naturalWidth,h=img.naturalHeight;
      if(first){
        pdf.addPage([w,h],'png');
        pdf.addImage(dataUrl,'PNG',0,0,w,h);
        first=false;
      }else{
        pdf.addPage([w,h],'png');
        pdf.addImage(dataUrl,'PNG',0,0,w,h);
      }
      console.log('Pagina '+(i+1)+' de '+arr.length+' processada.');
    }catch(e){
      console.warn('Falha na pagina '+(i+1)+': '+e.message);
    }
  }

  /* Remove a primeira pagina em branco que o jsPDF cria automaticamente */
  if(pdf.getNumberOfPages()>1){
    pdf.deletePage(1);
  }

  pdf.save('documento_baixado.pdf');
  console.log('Pronto! PDF salvo com '+arr.length+' paginas.');
})();
