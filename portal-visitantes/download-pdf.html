Baixar PDF do Google Drive (Somente Leitura)

Ferramenta para baixar PDFs que estão em modo visualização apenas

Use esta ferramenta apenas em documentos que você tem permissão legítima de visualização. O conteúdo já está visível na sua tela — este script apenas salva localmente o que você já pode ver.
Método 1: Console
Método 2: Bookmarklet
1

Abra o PDF no Google Drive

Acesse o link do PDF e aguarde carregar completamente
2

Abra o Console do navegador

Pressione F12 ou Ctrl+Shift+J (Chrome) / Ctrl+Shift+K (Firefox)
3

Cole o script abaixo no Console e pressione Enter

O script irá rolar o documento, coletar todas as páginas e baixá-las como imagens PNG 
(async function(){
  const urls = new Set();
  const sleep = ms => new Promise(r => setTimeout(r, ms));

  function collect() {
    document.querySelectorAll('img').forEach(img => {
      const s = img.src || '';
      if ((s.includes('googleusercontent') || s.includes('lh3') || s.includes('ggpht')) && img.naturalWidth > 200) {
        urls.add(s);
      }
    });
    document.querySelectorAll('[style*="background"]').forEach(el => {
      const bg = getComputedStyle(el).backgroundImage;
      if (bg && bg.includes('url(')) {
        const m = bg.match(/url\(["']?(.*?)["']?\)/);
        if (m && (m[1].includes('googleusercontent') || m[1].includes('lh3'))) urls.add(m[1]);
      }
    });
  }

  const container = document.querySelector('.ndfHFb-c4YZDc-q77wJc')
    || document.querySelector('[role="document"]')
    || document.body;
  let lastH = 0;

  for (let i = 0; i < 200; i++) {
    collect();
    container.scrollBy(0, window.innerHeight * 0.8);
    await sleep(700);
    if (container.scrollHeight === lastH && i > 3) break;
    lastH = container.scrollHeight;
  }

  container.scrollTo(0, 0);
  await sleep(500);
  collect();

  const arr = [...urls].sort();
  console.log('Encontradas ' + arr.length + ' paginas. Baixando...');

  arr.forEach((url, i) => {
    setTimeout(() => {
      fetch(url).then(r => r.blob()).then(blob => {
        const a = document.createElement('a');
        a.href = URL.createObjectURL(blob);
        a.download = 'pagina_' + String(i+1).padStart(3,'0') + '.png';
        a.click();
        URL.revokeObjectURL(a.href);
      }).catch(() => {
        const a = document.createElement('a');
        a.href = url;
        a.download = 'pagina_' + String(i+1).padStart(3,'0') + '.png';
        a.target = '_blank';
        a.click();
      });
    }, i * 400);
  });
})();

2

Abra o PDF no Google Drive e clique no favorito

O script fará o scroll automático e baixará as páginas
Observações:
  • O script baixa cada página como imagem PNG (não como PDF unificado)
  • O scroll automático pode levar alguns minutos em documentos longos
  • Não feche a aba durante o processo
  • Para juntar as PNGs em um único PDF, use um site como ilovepdf.com ou o comando convert *.pdf no terminal
